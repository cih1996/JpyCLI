# 当前项目里没看到“直接调用中间件接口刷机”的独立命令，现有刷机能力走 COM 口方向

# COM口刷机方向

# 上传本地刷机包到远程机器
go run . file push ./rom.zip --remote 192.168.1.100:9090 --dest D:\flash\rom.zip

# 让远程机器从 URL 下载刷机包
go run . file pull "https://example.com/rom.zip" --remote 192.168.1.100:9090 --dest D:\flash\rom.zip

# 单通道刷机
go run . flash run --com COM3 --ch 1 --mw 172.25.0.251 --ip-start 172.25.0.11 --script "C:/ai-services/rom/8se-20260309/002.cmd" -y

# 多通道刷机
go run . flash run --com COM3 --ch 1-10 --mw 172.25.0.251 --ip-start 172.25.0.11 --script "C:/ai-services/rom/8se-20260309/002.cmd" -y

# 指定多个通道刷机
go run . flash run --com COM3 --ch 1,3,5,7 --mw 172.25.0.251 --ip-start 172.25.0.11 --script "C:/ai-services/rom/8se-20260309/002.cmd" -y

# 多 COM 口批量刷机
go run . flash run --com COM3,COM4 --ch 1-20 --mw 172.25.0.251 --ip-start 172.25.0.11 --script "C:/ai-services/rom/8se-20260309/002.cmd" -y

# 所有 COM 口批量刷机
go run . flash run --com all --ch 1-20 --mw 172.25.0.251 --ip-start 172.25.0.11 --script "C:/ai-services/rom/8se-20260309/002.cmd" -y

# 模拟运行，先看任务分配
go run . flash run --com COM3 --ch 1-5 --mw 172.25.0.251 --ip-start 172.25.0.11 --script "C:/ai-services/rom/8se-20260309/002.cmd" --dry

# 指定超时和重试次数
go run . flash run --com COM3 --ch 1-10 --mw 172.25.0.251 --ip-start 172.25.0.11 --script "C:/ai-services/rom/8se-20260309/002.cmd" --timeout 1800 --retry 2 -y

# 远程刷机
go run . --remote 192.168.1.100:9090 flash run --com COM3 --ch 1-10 --mw 172.25.0.251 --ip-start 172.25.0.11 --script "C:/ai-services/rom/8se-20260309/002.cmd" -y

# 远程异步刷机
go run . --remote 192.168.1.100:9090 flash run --com COM3 --ch 1-20 --mw 172.25.0.251 --ip-start 172.25.0.11 --script "C:/ai-services/rom/8se-20260309/002.cmd" -y --async --async-timeout 1800

# 查询远程异步刷机任务
go run . shell --remote 192.168.1.100:9090 --task <task_id>

# 终止远程异步刷机任务
go run . shell --remote 192.168.1.100:9090 --kill <task_id>
